\begin{enumerate}
\setbeamertemplate{enumerate items}{}
\itemsep=1.5em\setstretch{1.15}
\setlength{\itemindent}{-1.25em}
\item \textbf{Filter Bubbles}
	\begin{enumerate}
	\vspace{.5em}
	\setbeamertemplate{enumerate items}{-}
	\itemsep=.8em\setstretch{1}
	\setlength{\itemindent}{-3em}
	\normalsize
	\item \textbf{\color{mycolor1}No Recommendation}: filter bubble effect; negative correlation between welfare and diversity
	\item \textbf{\color{mycolor2}Recommendation}: \textit{mitigates filter bubbles}; no correlation between welfare and diversity
	\end{enumerate}
\item \textbf{User Homogeneity}
	\begin{enumerate}
	\vspace{.5em}
	\setbeamertemplate{enumerate items}{-}
	\itemsep=.8em\setstretch{1}
	\setlength{\itemindent}{-3em}
	\normalsize
	\item \textbf{\color{mycolor1}No Recommendation}: lower coordination than {\color{mycolor3}optimal} ({\color{mycolor3}oracle}); idiosyncratic consumption paths 
	\item \textbf{\color{mycolor2}Recommendation}: higher coordination than {\color{mycolor3}optimal}; over-exploitation of other users' preferences
	\end{enumerate}

\item Retailers often sell their \textbf{own products}
	\begin{enumerate}
	\vspace{.5em}
	\setbeamertemplate{enumerate items}{-}
	\itemsep=.8em\setstretch{1}
	\setlength{\itemindent}{-1em}

	\item Standard in-house product development
	\item Many retailers have their own private clothings brands
	\end{enumerate}
\pause
\item Online platforms and \textbf{peak cloning}
	\begin{enumerate}
	\vspace{.5em}
	\setbeamertemplate{enumerate items}{-}
	\itemsep=.8em\setstretch{1}
	\setlength{\itemindent}{-1em}
	\normalsize
	\item Retailers already able to clone and substitute more homogeneous goods that were sure bets (e.g. Walmart's Dr. Pepper soda clone called Dr. Thunder)
	\item Detailed information allows online platforms/retailers better information on demand for attributes and product appeal and to clone rising and more successful products

	\item[] \textit{Amazon's advantage over traditional retailers is its knowledge and access to data from its platform}. (NYT, 2018.06.23)
	\end{enumerate}
